class Hero(SQLModel, table=True):
    id: int|None = Field(default=None, primary_key=True)
    name: str
    secret_name: str
    age: int|None = None

def select_heroes():
    with Session(engine) as session:
        statement = select(Hero).where(col(Hero.age) >= 35)
        results = session.exec(statement)
        for hero in results:
            print(hero)
