primes = []

st = [False] * 1000000

def get_prime(n):
    for i in range(2,n+1):
        if st[i]==False:
            primes.append(i)
        j= 0
        while(primes[j]*i<=n):
            st[primes[j]*i] = True
            if i%primes[j]==0: 
                break
            j += 1

get_prime(100)

日期的偏差

timedelta

from datetime import date, timedelta
d = date(2023, 3, 31)
print(d + timedelta(days=1)) # 2023-04-01
t = timedelta(days=1,hours=1,minutes=1,seconds=1) # 1 day, 1:01:01
print(t.total_seconds()) # 90061.0
print(t.days) # 1
print(t.seconds) # 3661 可以得到所有的秒数

import heapq
a = [1, 3, 5, 7, 9]
heapq.heapify(a) # 堆化
print(a) # [1, 3, 5, 7, 9]
heapq.heappush(a, 2) # 插入
print(a) # [1, 2, 5, 7, 9]
#pop 
heapq.heappop(a) # 弹出最小值
print(a) # [2, 3, 5, 7, 9]
